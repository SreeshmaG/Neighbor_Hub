            // DOM elements
            const categoryCards = document.querySelectorAll('.category-card');
            const locationCards = document.querySelectorAll('.location-card');
            const providerList = document.getElementById('providerList');
            const sortSelect = document.getElementById('sortSelect');
            const searchInput = document.querySelector('.search-bar input');
            const searchButton = document.querySelector('.search-bar button');

            // Selected service and area
            let selectedService = null;
            let selectedArea = null;

            // Event listeners for category cards
            categoryCards.forEach(card => {
                card.addEventListener('click', function() {
                    // Remove selected class from all cards
                    categoryCards.forEach(c => c.classList.remove('selected'));
                    
                    // Add selected class to clicked card
                    this.classList.add('selected');
                    
                    // Set selected service
                    selectedService = this.getAttribute('data-service');
                    
                    // If area is already selected, show providers
                    if (selectedArea) {
                        showProviders(selectedService, selectedArea);
                    }
                });
            });

            // Event listeners for location cards
            locationCards.forEach(card => {
                card.addEventListener('click', function() {
                    // Remove selected class from all cards
                    locationCards.forEach(c => c.classList.remove('selected'));
                    
                    // Add selected class to clicked card
                    this.classList.add('selected');
                    
                    // Set selected area
                    selectedArea = this.getAttribute('data-area');
                    
                    // If service is already selected, show providers
                    if (selectedService) {
                        showProviders(selectedService, selectedArea);
                    }
                });
            });

            // Sort functionality
            sortSelect.addEventListener('change', function() {
                if (selectedService && selectedArea) {
                    showProviders(selectedService, selectedArea);
                }
            });

            // Search functionality
            searchButton.addEventListener('click', function() {
                const searchTerm = searchInput.value.toLowerCase();
                if (searchTerm && selectedService && selectedArea) {
                    filterProviders(searchTerm);
                }
            });

            // Function to show providers
            function showProviders(service, area) {
                const areaProviders = providers[service][area];
                if (!areaProviders || areaProviders.length === 0) {
                    providerList.innerHTML = `<p class="no-providers">No providers available for ${service} in ${area}.</p>`;
                    return;
                }

                // Sort providers based on selected option
                const sortedProviders = [...areaProviders];
                const sortBy = sortSelect.value;
                
                if (sortBy === 'rating') {
                    sortedProviders.sort((a, b) => b.rating - a.rating);
                } else if (sortBy === 'distance') {
                    sortedProviders.sort((a, b) => parseFloat(a.distance) - parseFloat(b.distance));
                } else if (sortBy === 'price') {
                    sortedProviders.sort((a, b) => {
                        const priceA = parseFloat(a.price.replace(/[^\d.]/g, ''));
                        const priceB = parseFloat(b.price.replace(/[^\d.]/g, ''));
                        return priceA - priceB;
                    });
                }

                // Generate provider cards
                providerList.innerHTML = sortedProviders.map(provider => `
                    <div class="provider-card">
                        <div class="provider-header">
                            <div class="provider-avatar">${provider.name.split(' ').map(n => n[0]).join('')}</div>
                            <div class="provider-info">
                                <h3>${provider.name}</h3>
                                <div class="provider-rating">
                                    ${generateStarRating(provider.rating)}
                                    <span>(${provider.reviews} reviews)</span>
                                </div>
                            </div>
                        </div>
                        <div class="provider-details">
                            <p><i class="fas fa-phone"></i> ${provider.phone}</p>
                            <p><i class="fas fa-rupee-sign"></i> ${provider.price}</p>
                            <p><i class="fas fa-map-marker-alt"></i> ${provider.distance} away</p>
                            ${provider.experience ? `<p><i class="fas fa-briefcase"></i> ${provider.experience} experience</p>` : ''}
                            ${provider.subjects ? `<p><i class="fas fa-book"></i> Subjects: ${provider.subjects}</p>` : ''}
                        </div>
                        <div class="provider-actions">
                            <button class="action-btn call-btn" onclick="callProvider('${provider.phone}')">
                                <i class="fas fa-phone"></i> Call
                            </button>
                            <button class="action-btn whatsapp-btn" onclick="whatsappProvider('${provider.phone}')">
                                <i class="fab fa-whatsapp"></i> WhatsApp
                            </button>
                        </div>
                    </div>
                `).join('');
            }

            // Function to generate star rating
            function generateStarRating(rating) {
                let stars = '';
                const fullStars = Math.floor(rating);
                const hasHalfStar = rating % 1 >= 0.5;
                
                for (let i = 1; i <= 5; i++) {
                    if (i <= fullStars) {
                        stars += '<i class="fas fa-star"></i>';
                    } else if (i === fullStars + 1 && hasHalfStar) {
                        stars += '<i class="fas fa-star-half-alt"></i>';
                    } else {
                        stars += '<i class="far fa-star"></i>';
                    }
                }
                
                return stars;
            }

            // Function to filter providers by search term
            function filterProviders(searchTerm) {
                const providerCards = document.querySelectorAll('.provider-card');
                let hasResults = false;
                
                providerCards.forEach(card => {
                    const providerName = card.querySelector('h3').textContent.toLowerCase();
                    if (providerName.includes(searchTerm)) {
                        card.style.display = 'block';
                        hasResults = true;
                    } else {
                        card.style.display = 'none';
                    }
                });
                
                if (!hasResults) {
                    providerList.innerHTML = `<p class="no-results">No providers match your search for "${searchTerm}".</p>`;
                }
            }

            };
